# First installation
- Download image
    from http://cdimage.ubuntu.com/releases/
    ubuntu-18.04.2-preinstalled-server-arm64+raspi3.img.xz
- Burn image using 'etcher'
# With USB to TTL
- Connect USB to TTL : - - Black White Green
- sudo screen /dev/ttyUSB0 115200
- boot 3A+ with burned SD
- You must see boot on screen.
    Wait until login
    ubuntu/ubuntu
# Without USB to TTL
- Wait long enough so boot complete ~ 5min
- Remove SD and connect SD to the PC
- Change /etc/netplan/50.cloud-init.yaml as under

# Update Wifi connection, using netplan (18.04)
# Example with cat /usr/share/doc/netplan.io/examples/wireless.yaml
- edit /etc/netplan/xxx
- move 'ethernets' to wifis, update to a wifi like entry with access-points and gateway4
- sudo netplan apply

ssh ubuntu@ubuntu

# Camera and Pi camera tools
# The camera is directly connected to the gpu.
# Useful links :
#  https://raspberrypi.stackexchange.com/questions/37359/how-to-use-raspistill-on-ubuntu
#  https://picamera.readthedocs.io/en/release-1.13/index.html
- edit /boot/firmware/config.txt and add
start_x=1
gpu_mem=128
- install raspi-config
wget https://archive.raspberrypi.org/debian/pool/main/r/raspi-config/raspi-config_20180406+1_all.deb -P /tmp
apt-get install libnewt0.52 whiptail parted triggerhappy lua5.1 alsa-utils -y
apt-get install -fy
dpkg -i /tmp/raspi-config_20180406+1_all.deb
- install userland. Raspi camera drivers
sudo apt install cmake
git clone https://github.com/raspberrypi/userland.git
cd userland
./buildme
touch ~/.bash_aliases
echo -e 'PATH=$PATH:/opt/vc/bin\nexport PATH' >> ~/.bash_aliases
echo -e 'LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/opt/vc/lib\nexport LD_LIBRARY_PATH' >> ~/.bash_aliases
source ~/.bashrc
ldconfig
- give rights to camera
sudo sh -c 'echo "SUBSYSTEM==\"vchiq\",GROUP=\"video\",MODE=\"0660\"" > /etc/udev/rules.d/10-vchiq-permissions.rules'
sudo usermod -a -G video $(whoami)
reboot
- test with raspistill
raspistill -o cam.jpg
- install python picamera using pip
  Follow instruction in picamera read the docs.
sudo apt install python3-pip
